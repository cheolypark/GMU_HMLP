SELECT  

LatitudeReport,
LongitudeReport,
AltitudeReport,
DistanceToCIReport,
DirectionToCIReport,
DistanceToSensor,
Latitude,
Longitude,
Altitude,
DistanceToCI,
DirectionToCI 

FROM  mti_report as mr 
join  reportedtarget_mtirpt as rm 
join sensortemporalproperty as st 
join  targettemporalproperty as tt 

on tt.tr = st.tr and tt.t = st.t 
and tt.t = mr.t and rm.mtireportedtargetid = mr.rt_mti 
and rm.actualtarget_mtirpt = tt.tr
